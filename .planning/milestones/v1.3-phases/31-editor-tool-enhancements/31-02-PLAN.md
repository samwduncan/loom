    3. Create the handler function inside the component:
    ```typescript
    function handleRunInTerminal() {
      if (!command) return;
      useUIStore.getState().setActiveTab('shell');
      // rAF ensures tab switch CSS visibility triggers before send
      requestAnimationFrame(() => {
        const sent = sendToShell(command + '\n');
        if (!sent) {
          // Terminal not yet connected -- retry once after mount
          setTimeout(() => sendToShell(command + '\n'), 500);
        }
      });
    }
    ```

    4. Determine button visibility: show only when `status === 'resolved'` AND `command` is a non-empty string. Do NOT show for pending/running/error states.

    5. Add the action button as a footer row below the output body, inside the outer container div:

    {status === 'resolved' && command && (

        <Button
          variant="ghost"
          size="sm"
          className="h-6 px-2 text-xs gap-1 text-[var(--text-muted)] hover:text-[var(--text-primary)]"
          onClick={handleRunInTerminal}
        >
          <Play className="w-3 h-3" />
          Run in Terminal
        </Button>

    )}
