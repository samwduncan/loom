**1. Create `src/src/components/chat/view/LiveSessionBanner.tsx`:**

A simple banner that shows when the current session is live-attached.

```tsx
import { Radio } from 'lucide-react';
import { wsClient } from '@/lib/websocket-client';
import { useStreamStore } from '@/stores/stream';
import { cn } from '@/utils/cn';

interface LiveSessionBannerProps {
  sessionId: string;
}

export function LiveSessionBanner({ sessionId }: LiveSessionBannerProps) {
  const isAttached = useStreamStore((s) => s.liveAttachedSessions.has(sessionId));

  if (!isAttached) return null;

  const handleDetach = () => {
    wsClient.send({ type: 'detach-session', sessionId });
    // Optimistic: also update store immediately (server confirmation will follow)
    useStreamStore.getState().detachLiveSession(sessionId);
  };

  return (
    <div
      className={cn(
        'flex items-center gap-2 px-4 py-2',
        'bg-[oklch(0.25_0.02_142)] border-b border-[oklch(0.35_0.05_142)]',
        'text-[oklch(0.85_0.12_142)] text-sm',
      )}
      role="status"
      aria-live="polite"
    >
      <Radio size={14} className="text-[oklch(0.72_0.19_142)] animate-pulse" />
      <span className="flex-1">Watching live session</span>
      <button
        onClick={handleDetach}
        className={cn(
          'px-2 py-0.5 rounded text-xs',
          'bg-[oklch(0.3_0.02_142)] hover:bg-[oklch(0.35_0.02_142)]',
          'text-[oklch(0.75_0.08_142)] hover:text-[oklch(0.85_0.08_142)]',
          'transition-colors duration-[var(--duration-fast)]',
        )}
      >
        Detach
      </button>
    </div>
  );
}
```

**Add auto-attach useEffect (after the existing session loading useEffect):**

This checks if the session was recently active (JSONL file was recently modified) by checking the session's `updatedAt` timestamp. If the session was updated within the last 5 minutes and is not already attached, send an attach-session message.

```typescript
// Auto-attach to live sessions: if session's last activity is within 5 minutes, attempt attach
useEffect(() => {
  if (!sessionId || !projectName || sessionId.startsWith('stub-')) return;

  // Don't auto-attach if already attached
  const streamState = useStreamStore.getState();
  if (streamState.liveAttachedSessions.has(sessionId)) return;

  // Check if session appears "active" based on recent updatedAt
  const session = useTimelineStore.getState().sessions.find(s => s.id === sessionId);
  if (!session) return;

  const lastActivity = new Date(session.updatedAt).getTime();
  const fiveMinutesAgo = Date.now() - 5 * 60 * 1000;

  if (lastActivity > fiveMinutesAgo) {
    wsClient.send({
      type: 'attach-session',
      sessionId,
      projectName,
    });
  }

  // Auto-detach when navigating away from this session
  return () => {
    if (useStreamStore.getState().liveAttachedSessions.has(sessionId)) {
      wsClient.send({ type: 'detach-session', sessionId });
      useStreamStore.getState().detachLiveSession(sessionId);
    }
  };
}, [sessionId, projectName]);
```

```typescript
// 1.5. Detach from any live-attached session
const currentLiveAttached = useStreamStore.getState().liveAttachedSessions;
for (const attachedId of currentLiveAttached) {
  wsClient.send({ type: 'detach-session', sessionId: attachedId });
}
if (currentLiveAttached.size > 0) {
  // Clear all live attachments in store (we're switching away)
  for (const attachedId of currentLiveAttached) {
    useStreamStore.getState().detachLiveSession(attachedId);
  }
}
```

This ensures that switching from a live-attached session to another session sends the detach message immediately, before the new session is loaded. This prevents orphaned file watchers on the backend.

Note: The ChatView cleanup useEffect would also fire, but this provides earlier cleanup in the switch flow for a snappier experience (no race between navigate and cleanup).
