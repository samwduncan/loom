Create `server/cache/session-watcher.js` as an ES module exporting a `SessionWatcher` class that extends `EventEmitter`.

**Constructor:**
- `this.watched = new Map()` -- maps sessionId -> `{ filePath, offset, watcher, buffer }`
- `this.MAX_WATCHES = 5` -- hard cap on simultaneous watches

**`watch(sessionId, filePath)` method:**
- If already watching this sessionId, return (idempotent)
- If at MAX_WATCHES capacity, throw an Error: "Maximum simultaneous watches reached"
- Get initial file size via `fs.statSync(filePath).size` as the starting byte offset
- Create an `fs.watch(filePath)` watcher listening for `'change'` events
- On change, call `this._readNewLines(sessionId)` with a 100ms debounce per session (use a per-session setTimeout stored in the map entry as `debounceTimer`)
- Store `{ filePath, offset, watcher, buffer: '', debounceTimer: null }` in the map keyed by sessionId

**`unwatch(sessionId)` method:**
- Get entry from map, if not found return
- Close watcher via `entry.watcher.close()`
- Clear debounceTimer if set
- Delete from map
- Emit `'unwatched'` event with `{ sessionId }`

**`unwatchAll()` method:**
- Iterate map, call `unwatch()` for each

**`getActiveWatches()` method:**
- Return `Array.from(this.watched.keys())` -- list of watched sessionIds

**`isWatching(sessionId)` method:**
- Return `this.watched.has(sessionId)`

**`async _readNewLines(sessionId)` method (private):**
- Get entry from map; return if not found
- Get current file size via `fs.promises.stat(entry.filePath).then(s => s.size)` -- if file is smaller than offset (truncation), reset offset to 0
- If currentSize <= entry.offset, return (no new data)
- Create a ReadStream: `fs.createReadStream(entry.filePath, { start: entry.offset, encoding: 'utf8' })`
- Read all chunks into a string, prepend any leftover `entry.buffer`
- Split by `\n`. If the last element is not empty string (no trailing newline), store it in `entry.buffer` for next read (partial line buffering). Otherwise clear buffer.
- Parse each non-empty line as JSON. Wrap in try/catch per line -- skip lines that fail to parse (log warning).
- Update `entry.offset = currentSize`
- If parsed entries array is non-empty, emit `'entries'` event: `{ sessionId, entries }`

**Error handling:**
- Wrap `_readNewLines` in try/catch. On ENOENT (file deleted), call `this.unwatch(sessionId)` and emit `'error'` event: `{ sessionId, error: 'File deleted' }`.
- fs.watch error handler: log and unwatch.

**Module export:**
```javascript
// Singleton instance -- one watcher per server
const sessionWatcher = new SessionWatcher();
export default sessionWatcher;
export { SessionWatcher };
```

**4. Add attach-session handler in handleChatConnection message dispatch (after the `get-active-sessions` handler, around line 1089):**
```javascript
} else if (data.type === 'attach-session') {
    const { sessionId, projectName } = data;
    console.log('[INFO] Attach session request:', sessionId, 'project:', projectName);

    // Resolve project directory and find the JSONL file for this session
    try {
        const projectDir = path.join(os.homedir(), '.claude', 'projects', projectName);
        const sessionMeta = messageCache.getSessionMeta(sessionId);
        let jsonlFile = sessionMeta?.jsonl_file;

        if (!jsonlFile) {
            // Scan project directory for JSONL files containing this session
            const files = await fsPromises.readdir(projectDir);
            const jsonlFiles = files.filter(f => f.endsWith('.jsonl') && !f.startsWith('agent-'));
            // For simplicity, use the most recently modified .jsonl file as a heuristic
            // (sessions are typically in one file; the cache usually knows the right file)
            for (const f of jsonlFiles) {
                const content = await fsPromises.readFile(path.join(projectDir, f), 'utf8');
                if (content.includes(sessionId)) {
                    jsonlFile = f;
                    break;
                }
            }
        }

        if (!jsonlFile) {
            writer.send({ type: 'error', error: `Cannot find JSONL file for session ${sessionId}` });
            return;
        }

        const filePath = path.join(projectDir, jsonlFile);

        // Start watching (idempotent -- won't double-watch)
        if (!sessionWatcher.isWatching(sessionId)) {
            sessionWatcher.watch(sessionId, filePath);
        }

        // Track this client's attachment
        if (!clientAttachments.has(ws)) {
            clientAttachments.set(ws, new Set());
        }
        clientAttachments.get(ws).add(sessionId);

        writer.send({
            type: 'live-session-attached',
            sessionId,
            watching: true
        });
    } catch (err) {
        console.error('[ERROR] Attach session failed:', err.message);
        writer.send({ type: 'error', error: `Attach failed: ${err.message}` });
    }
} else if (data.type === 'detach-session') {
    const { sessionId } = data;
    console.log('[INFO] Detach session request:', sessionId);

    // Remove this client's attachment
    const attachments = clientAttachments.get(ws);
    if (attachments) {
        attachments.delete(sessionId);
    }

    // If no clients are watching this session anymore, stop the file watcher
    let anyClientWatching = false;
    for (const [, sessions] of clientAttachments.entries()) {
        if (sessions.has(sessionId)) {
            anyClientWatching = true;
            break;
        }
    }
    if (!anyClientWatching) {
        sessionWatcher.unwatch(sessionId);
    }

    writer.send({
        type: 'live-session-detached',
        sessionId
    });
}
```
