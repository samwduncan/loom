    **Step 1: Create `mobile/components/navigation/AnimatedScreen.tsx`** -- Parallax wrapper:
    ```typescript
    import { type PropsWithChildren } from 'react';
    import Animated, { useAnimatedStyle, interpolate } from 'react-native-reanimated';
    import { useDrawerProgress } from '@react-navigation/drawer';

    export function AnimatedScreen({ children }: PropsWithChildren) {
      const progress = useDrawerProgress();

      const animatedStyle = useAnimatedStyle(() => ({
        transform: [
          { translateX: interpolate(progress.value, [0, 1], [0, 20]) },
        ],
      }));

      return (
        <Animated.View style={[{ flex: 1 }, animatedStyle]}>
          {children}
        </Animated.View>
      );
    }
    ```

    Per RESEARCH Pitfall 2 (Double-Shift): `drawerType: 'slide'` already shifts the scene. This wrapper adds an ADDITIONAL 20px parallax on top. If testing reveals the total shift is too much, adjust the interpolation output range from `[0, 20]` to a smaller value. The important thing is the parallax EFFECT exists -- exact pixels can be tuned.
