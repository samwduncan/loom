```typescript
export interface AuthProvider {
  getToken(): string | null;
  setToken(token: string): void;
  clearToken(): void;
}
```

API client factory (per research Pattern 2, with auth refresh callback):
```typescript
export interface ApiClientOptions {
  auth: AuthProvider;
  resolveUrl: (path: string) => string;
  onAuthRefresh?: () => Promise<string | null>;
}

export function createApiClient(options: ApiClientOptions) {
  async function apiFetch<T>(path: string, options?: RequestInit, signal?: AbortSignal): Promise<T> { ... }
  return { apiFetch };
}
```

WebSocket client factory (resolveWsUrl includes token parameter):
```typescript
export interface WebSocketClientOptions {
  resolveWsUrl: (path: string, token: string) => string;
  auth: AuthProvider;
}
```
NOTE: The existing `resolveWsUrl` in web platform.ts has signature `resolveWsUrl(path: string, token: string): string` -- it embeds the token in the WebSocket URL as a query parameter. The shared WebSocket client's `connect()` calls `this.auth.getToken()` and passes the token to `this.resolveWsUrl(path, token)`.

Web AuthProvider implementation:
```typescript
import type { AuthProvider } from '@loom/shared/lib/auth';
const TOKEN_KEY = 'loom-jwt';
export const webAuthProvider: AuthProvider = {
  getToken: () => localStorage.getItem(TOKEN_KEY),
  setToken: (token: string) => localStorage.setItem(TOKEN_KEY, token),
  clearToken: () => localStorage.removeItem(TOKEN_KEY),
};
```

Web store wrapper pattern:
```typescript
import { createTimelineStore } from '@loom/shared/stores/timeline';
const localStorageAdapter = {
  getItem: (name: string) => localStorage.getItem(name),
  setItem: (name: string, value: string) => localStorage.setItem(name, value),
  removeItem: (name: string) => localStorage.removeItem(name),
};
export const useTimelineStore = createTimelineStore(localStorageAdapter);
if (import.meta.env.DEV) {
  (window as unknown as Record<string, unknown>).__ZUSTAND_TIMELINE_STORE__ = useTimelineStore;
}
```
