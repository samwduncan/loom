**ReconnectSkeletons.tsx:**

```typescript
import { useState, useEffect } from 'react';
import '../../styles/aurora-shimmer.css';

interface ReconnectSkeletonsProps {
  count: number;         // Number of skeleton message placeholders
  isLoading: boolean;    // True during reconnect load, false when messages arrive
}
```

Component renders `count` skeleton message placeholders, each containing 2-3 aurora skeleton lines. When `isLoading` transitions to false, the skeletons fade out (200ms opacity transition) before unmounting:

```tsx
export default function ReconnectSkeletons({ count, isLoading }: ReconnectSkeletonsProps) {
  const [visible, setVisible] = useState(true);
  const [fading, setFading] = useState(false);

  useEffect(() => {
    if (!isLoading && visible) {
      setFading(true);
      const timer = setTimeout(() => setVisible(false), 200);
      return () => clearTimeout(timer);
    }
    if (isLoading) {
      setVisible(true);
      setFading(false);
    }
  }, [isLoading]);

  if (!visible) return null;

  return (
    <div style={{ opacity: fading ? 0 : 1, transition: 'opacity 200ms ease-out' }}>
      {Array.from({ length: count }).map((_, i) => (
        <div key={i} className="py-3 px-4 space-y-2">
          <div className="aurora-skeleton-line" style={{ width: '100%' }} />
          <div className="aurora-skeleton-line" style={{ width: '90%' }} />
          <div className="aurora-skeleton-line" style={{ width: '75%' }} />
        </div>
      ))}
    </div>
  );
}
```

Note: This uses fade-in transition (opacity 200ms) per user decision — distinct from the collapse-upward used for PreTokenIndicator. The user explicitly requested two different transition styles for different contexts.
