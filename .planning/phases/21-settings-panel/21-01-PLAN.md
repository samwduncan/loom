     ```typescript
       codeFontFamily: string;
     }
     ```

  4. Create `src/src/hooks/useSettingsData.ts` with per-tab data fetching hooks. All use `apiFetch<T>` from `@/lib/api-client`. Each hook returns `{ data, isLoading, error, refetch }` pattern:
     - `useAgentStatuses()`: Fetches from `/api/cli/claude/status`, `/api/cli/codex/status`, `/api/cli/gemini/status` in parallel via Promise.all. Returns `ProviderStatus[]`.
     - `useApiKeys()`: Fetches from `/api/settings/api-keys`. Returns `ApiKeyResponse[]`. Includes `addKey(keyName: string)` (POST), `deleteKey(id: number)` (DELETE), `toggleKey(id: number, isActive: boolean)` (PATCH).
     - `useCredentials()`: Fetches from `/api/settings/credentials`. Returns `CredentialResponse[]`. Includes `addCredential(data)` (POST), `deleteCredential(id: number)` (DELETE).
     - `useGitConfig()`: Fetches from `/api/user/git-config`. Returns `GitConfigResponse`. Includes `saveGitConfig(gitName, gitEmail)` (POST).
     - `useMcpServers(provider: 'claude' | 'codex')`: Fetches from `/api/mcp/config/read` (claude) or `/api/codex/mcp/config/read` (codex). Returns `McpServer[]`. Includes `addServer(data)` (POST), `removeServer(name: string)` (DELETE).
     Use AbortController for cleanup on unmount. Fetch on mount via useEffect. Mutation functions should call refetch() after success and return a promise so callers can toast.
