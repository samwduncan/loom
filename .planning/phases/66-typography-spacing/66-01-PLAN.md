```css
/* -- Mobile Typography Overrides (TYPO-04, TYPO-07) ----------------
 * Mobile reading at arm's length benefits from slightly larger body
 * text (15px vs 14px desktop) and significantly larger code (18px vs
 * 13px desktop). Line-height 1.6 improves readability for multi-line
 * content. Both .markdown-body (finalized) and .active-message
 * (streaming) MUST match to prevent CLS on stream completion.
 * Per D-02: No clamp() -- clean breakpoint split only.
 */
@media (max-width: 767px) {
  .markdown-body,
  .active-message {
    font-size: 0.9375rem;   /* 15px body text at mobile distance (TYPO-04) */
    line-height: 1.6;       /* TYPO-07: mobile reading comfort */
  }

  .markdown-body code,
  .markdown-body pre code,
  .markdown-body [data-highlighted] code,
  .active-message code,
  .active-message .streaming-code-block code {
    font-size: 1.125rem;    /* 18px code at mobile distance (TYPO-04) */
  }

  /* TYPO-06: Action labels minimum 14px on mobile (per D-16, D-17) */
  [cmdk-item],
  [cmdk-empty] {
    font-size: 0.875rem;    /* 14px -- already this value, defensive floor */
  }
}
```
